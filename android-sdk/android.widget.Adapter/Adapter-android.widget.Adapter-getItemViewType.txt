自定义，对应位置的数据，对应的视图类型

	@Override
	public int getItemViewType(int position) {
		HelpItemData data = mDataSource.get(position);
		if (data != null) {
			if (data.type == 1) {
				return 1;
			} else if (data.type == 2) {
				return 2;
			} else if (data.type == 3) {
				return 3;
			}
		}
		return 0;
	}
