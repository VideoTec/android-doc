代码：    private static Handler sMediaHandler;    private static Handler sControlHandler;    public static Handler getMediaHandler() {        return getAsyncHandler(sMediaHandler, VOIP_MEDIA_THREAD);    }    public static Handler getControlHandler() {        return getAsyncHandler(sControlHandler, VOIP_CONTROL_THREAD);    }    private static synchronized Handler getAsyncHandler(Handler h, String threadName) {        if (h == null) {            HandlerThread asyncThread =                    new HandlerThread(threadName,                            android.os.Process.THREAD_PRIORITY_BACKGROUND);            asyncThread.start();            h = new Handler(asyncThread.getLooper());        }        return h;    }    警告：Private field 'sMediaHandler' is never assigned
